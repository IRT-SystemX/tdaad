import numpy as np
import pandas as pd

from tdaad.twe_logger import get_logger

logger = get_logger()


def _to_timestamp_eval(index):
    """Modified python builtin `eval` to interpret `"Timestamp"` string as a pandas Timestamp."""
    try:
        # /!\ some hack we got here /!\, danger?
        converted_index = eval(index, {
            "numpy": np,
            "Timestamp": pd._libs.tslibs.timestamps.Timestamp,
        })
    except NameError as exc:
        logger.info("Could not convert tuples in string format to actual Timestamp tuples.")
        raise NameError from exc
    return converted_index


def _convert_indices_from_string_to_tuple(indices):
    """Map a series of string to their corresponding tuples."""
    tuple_mapping = indices.map(_to_timestamp_eval)
    return tuple_mapping.to_flat_index()

def simple_index_remapping(X):
    """From a dataframe with tuple index, create a new dataframe with index chosen as the last element from tuple."""
    tuple_indices = _convert_indices_from_string_to_tuple(X.index)
    tuple_index_lengths = tuple_indices.map(len)
    if not np.all(tuple_index_lengths == tuple_index_lengths[0]):
        logger.info("Tuples do not have the same length, could not figure out how to remap this feature.")
        return Exception

    X_new = X.copy()
    X_new['newidx'] = [idx[-1] for idx in tuple_indices]
    X_new.set_index("newidx", inplace=True)
    X_new.index.name = None
    return X_new

def nonbackward_index_remapping(X):
    """Simple remapping + interpolation. From a dataframe with tuple index, create a new dataframe with index chosen
    as the last element from tuple. Then map the original indices onto those and interpolate in between missing values.
    """

    tuple_indices = _convert_indices_from_string_to_tuple(X.index)
    flattened_tuples = np.unique(sum([list(tup) for tup in tuple_indices.values], []))

    X_new = X.copy()
    X_new['newidx'] = [idx[-1] for idx in tuple_indices]
    X_new.set_index("newidx", inplace=True)
    X_new.index.name = None

    X_newer = pd.DataFrame(index=[idx for idx in flattened_tuples if idx not in X_new.index], columns=X_new.columns)

    X_better = pd.concat([X_new, X_newer]).sort_index()
    # interpolation should go forward so as to not infer from the future
    X_stronger = X_better.interpolate(limit_direction='forward')
    return X_stronger

def exhaustive_index_remapping(X):
    """Simple remapping + interpolation. From a dataframe with tuple index, create a new dataframe with index chosen
    as the last element from tuple. Then map the original indices onto those and interpolate in between missing values.
    """
    X_stronger = nonbackward_index_remapping(X)
    X_faster = X_stronger.fillna(method="bfill")
    return X_faster




def _majority_vote_index_remapping(X):
    """From a dataframe with tuple index, create dataframe views with index chosen as the nth element from tuple,
    then average (majority vote) those views.
    Careful: can be time & space consuming if length of tuples is large enough.
    """
    tuple_indices = _convert_indices_from_string_to_tuple(X.index)
    tuple_index_lengths = tuple_indices.map(len)
    if not np.all(tuple_index_lengths == tuple_index_lengths[0]):
        logger.info("Tuples do not have the same length, could not figure out how to remap this feature.")
        return Exception
    window_size = tuple_index_lengths[0]

    X_new = X.copy()
    if isinstance(X_new, pd.Series):
        X_new = X_new.to_frame()
    X_convert = X_new.set_index(tuple_indices)
    # for all position w in window tuples we consider dataframe with index tuple_indices[w]
    X_convert_mapped = [X_convert.set_index(X_convert.index.map(lambda x: x[w])) for w in range(window_size)]
    X_convert_mapped_concat = pd.concat(X_convert_mapped, axis=0)
    # all is left is to average the duplicated index value
    res = X_convert_mapped_concat.groupby(X_convert_mapped_concat.index)
    return res

def majority_vote_index_remapping(X):
    """Majority vote (see above) + mean."""
    res = _majority_vote_index_remapping(X).mean()
    return res

def get_index_remapping_fun(remapping_name):
    """ Getter for access to `INDEX_REMAPPING_DICT`"""
    if remapping_name not in INDEX_REMAPPING_DICT.keys():
        print(f"Index remapping function {remapping_name} not found, using simple index remapping.")
    return INDEX_REMAPPING_DICT.get(remapping_name, simple_index_remapping)

def score_flat_fast_remapping(scores, window_size, stride, padding_length=0):
    """Univariate score remapping without window indices and with precomputed padding_length."""
    if hasattr(scores, "index"):
        scores = scores.values
    begins = np.array([i * stride for i in range(scores.shape[0])])
    ends = begins + window_size

    remapped_scores = np.full(shape=stride * (scores.shape[0] - 1) + window_size + padding_length, fill_value=np.nan)

    # iterate over window intersection points, interpolate (vote) in-between intersection and next_intersection
    intersections = np.unique(np.r_[begins, ends])
    for intersection, next_intersection in zip(intersections[:-1], intersections[1:]):
        window_indices = np.flatnonzero((begins <= intersection) & (next_intersection < ends + 1))
        remapped_scores[intersection:next_intersection] = np.nansum(scores[window_indices])    # or nanmedian

    # replace unknown values with 0 = normal (for the padding at the end)
    np.nan_to_num(remapped_scores, copy=False)
    return remapped_scores
